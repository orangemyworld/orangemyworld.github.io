                              // Setup
                              void setup() {
                              //- Column pinMode
                                pinMode(latchPin1, OUTPUT);
                                pinMode(clockPin1, OUTPUT);
                                pinMode(dataPin1, OUTPUT);

                              //---- Row pinMode
                                pinMode(latchPin2, OUTPUT);

                              //|      Refresh

                              int RefreshDisplay(int zone1, int zone2, int zone3,int zone4, int zone5, int zone6,int zone7, int zone8, int zone9, int row){

                                digitalWrite(latchPin2, LOW);
                                shiftOut(dataPin2, clockPin2, MSBFIRST, row);

                                digitalWrite(latchPin1, LOW);
                                    shiftOut(dataPin1, clockPin1, LSBFIRST, zone9);
                                    shiftOut(dataPin1, clockPin1, LSBFIRST, zone8);
                                    shiftOut(dataPin1, clockPin1, LSBFIRST, zone7);
                                    shiftOut(dataPin1, clockPin1, LSBFIRST, zone6);
                                    shiftOut(dataPin1, clockPin1, LSBFIRST, zone5);
                                    shiftOut(dataPin1, clockPin1, LSBFIRST, zone4);
                                    shiftOut(dataPin1, clockPin1, LSBFIRST, zone3);
                                    shiftOut(dataPin1, clockPin1, LSBFIRST, zone2);
                                    shiftOut(dataPin1, clockPin1, LSBFIRST, zone1);

                                digitalWrite(latchPin1, HIGH);
                                digitalWrite(latchPin2, HIGH);
                                delayMicroseconds(500);
                              }
